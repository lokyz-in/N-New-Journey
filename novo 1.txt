// script.js

function atualizarStatus(fisico, velocidade, resistencia) {
    document.getElementById("statusFisico").innerHTML = fisico + "kg";
    document.getElementById("statusVelocidade").innerHTML = velocidade + "km/h";
    document.getElementById("statusResistencia").innerHTML = resistencia + "kg";
}

let fisico = 1000;
let velocidade = 300;
let resistencia = 1000;

document.getElementById("atualizarBtn").addEventListener("click", function() {
    fisico += 100; // Atualizar força
    velocidade += 50;  // Atualizar velocidade
    resistencia += 100;  // Atualizar resistência
    atualizarStatus(fisico, velocidade, resistencia);
});

const express = require('express');
const app = express();
const mysql = require('mysql2');

});

    });
});
